    Skeleton (main pieces):
    ```cpp
    /*
        JamWide Plugin - test_peer_churn_simulation.cpp
        Automated NINJAM-like peer-churn simulation per Codex M-5.
        Satisfies the automated-coverage portion of ROADMAP-15.1-S1.
    */
    #include "src/threading/spsc_ring.h"
    #include "src/threading/spsc_payloads.h"

    #include <atomic>
    #include <cstdio>
    #include <cstring>
    #include <thread>
    #include <variant>

    static int tests_run = 0;
    static int tests_passed = 0;
    #define TEST(name) do { tests_run++; printf("  TEST: %s ... ", name); fflush(stdout); } while(0)
    #define PASS()     do { tests_passed++; printf("PASSED\n"); } while(0)
    #define FAIL(msg)  do { printf("FAILED: %s\n", msg); } while(0)

    namespace simjamwide {
        class DecodeState   { public: int marker = 0; };

        constexpr int MAX_PEERS = 64;
        constexpr int MAX_USER_CHANNELS = 8;
        constexpr int MAX_LOCAL_CHANNELS = 32;

        struct RemoteUserChannelMirror { bool present = false; float volume = 1.0f; DecodeState* ds = nullptr; };
        struct RemoteUserMirror { bool active = false; int user_index = 0; RemoteUserChannelMirror chans[MAX_USER_CHANNELS]; };
        struct LocalChannelMirror { bool active = false; float volume = 1.0f; };

        RemoteUserMirror   remote_mirror[MAX_PEERS];
        LocalChannelMirror local_mirror[MAX_LOCAL_CHANNELS];

        jamwide::SpscRing<jamwide::RemoteUserUpdate,   64> ru_q;
        jamwide::SpscRing<jamwide::LocalChannelUpdate, 32> lc_q;
        jamwide::SpscRing<jamwide::DecodeArmRequest, jamwide::ARM_REQUEST_CAPACITY> arm_q;
        jamwide::SpscRing<DecodeState*, jamwide::DEFERRED_DELETE_CAPACITY> dd_q;

        std::atomic<uint64_t> audio_drain_gen{0};
        std::atomic<uint64_t> ru_drops{0};
        std::atomic<uint64_t> lc_drops{0};
        std::atomic<uint64_t> arm_drops{0};
    }

    // simulated_audio_thread_drain() — drains ru_q + lc_q via std::visit; bumps audio_drain_gen.
    // simulated_run_thread_drain_arms() — drains arm_q, constructs DecodeState, publishes PeerNextDsUpdate.
    // simulated_run_thread_drain_dd() — drains dd_q.
    // (full implementations: see test_remote_user_mirror.cpp + test_local_channel_mirror.cpp patterns)

    static void test_peer_churn_simulation_under_tsan() {
        TEST("M-5 peer-churn simulation: 1000 cycles, 2 threads, TSan-clean");
        using namespace simjamwide;
        // Reset state.
        for (auto& m : remote_mirror) m = {};
        for (auto& m : local_mirror) m = {};

        std::atomic<bool> stop{false};
        std::thread audio([&]{
            while (!stop.load(std::memory_order_acquire)) {
                simulated_audio_thread_drain();
                std::this_thread::yield();
            }
            simulated_audio_thread_drain();
        });
        std::thread runner([&]{
            constexpr int kCycles = 1000;
            for (int i = 0; i < kCycles; ++i) {
                const int slot = i % MAX_PEERS;
                const int lc   = i % MAX_LOCAL_CHANNELS;

                if (!ru_q.try_push(jamwide::RemoteUserUpdate{jamwide::PeerAddedUpdate{slot, slot * 7}})) ru_drops.fetch_add(1);

                jamwide::PeerChannelMaskUpdate cm{}; cm.slot = slot; cm.chanpresentmask = 0x3;
                if (!ru_q.try_push(jamwide::RemoteUserUpdate{cm})) ru_drops.fetch_add(1);

                jamwide::PeerVolPanUpdate vp{}; vp.slot = slot; vp.volume = 1.0f;
                if (!ru_q.try_push(jamwide::RemoteUserUpdate{vp})) ru_drops.fetch_add(1);

                jamwide::DecodeArmRequest req{}; req.slot = slot; req.channel = i % MAX_USER_CHANNELS; req.slot_idx = 0;
                if (!arm_q.try_push(req)) arm_drops.fetch_add(1);
                simulated_run_thread_drain_arms();

                jamwide::LocalChannelAddedUpdate la{}; la.channel = lc; la.volume = 1.0f;
                if (!lc_q.try_push(jamwide::LocalChannelUpdate{la})) lc_drops.fetch_add(1);

                jamwide::LocalChannelMonitoringUpdate lm{}; lm.channel = lc; lm.set_volume = true; lm.volume = 0.5f;
                if (!lc_q.try_push(jamwide::LocalChannelUpdate{lm})) lc_drops.fetch_add(1);

                if (!ru_q.try_push(jamwide::RemoteUserUpdate{jamwide::PeerRemovedUpdate{slot}})) ru_drops.fetch_add(1);
                if (!lc_q.try_push(jamwide::LocalChannelUpdate{jamwide::LocalChannelRemovedUpdate{lc}})) lc_drops.fetch_add(1);

                simulated_run_thread_drain_dd();

                if (i % 100 == 0) std::this_thread::sleep_for(std::chrono::milliseconds(1));
            }
            stop.store(true, std::memory_order_release);
        });
        runner.join();
        audio.join();
        simulated_run_thread_drain_dd();

        if (ru_drops.load() == 0 && lc_drops.load() == 0 && arm_drops.load() == 0) PASS();
        else FAIL("drop counters non-zero");
    }

    int main() {
        printf("test_peer_churn_simulation — Codex M-5 automated\n");
        test_peer_churn_simulation_under_tsan();
        printf("\n%d/%d tests passed\n", tests_passed, tests_run);
        return tests_passed == tests_run ? 0 : 1;
    }
    ```
