2. Add codec combo widget after the bitrate combo (around line 72, before the Transmit checkbox). Add:
   ```cpp
   // Codec selector (CODEC-03)
   ImGui::SameLine();
   const char* kCodecLabels[] = { "FLAC (Lossless)", "Vorbis (Compressed)" };
   ImGui::SetNextItemWidth(150.0f);
   if (ImGui::Combo("##codec_local", &state.local_codec_index,
                     kCodecLabels, 2)) {
       if (state.status == NJClient::NJC_STATUS_OK) {
           jamwide::SetEncoderFormatCommand cmd;
           cmd.fourcc = (state.local_codec_index == 0)
               ? NJ_ENCODER_FMT_FLAC
               : MAKE_NJ_FOURCC('O','G','G','v');
           plugin->cmd_queue.try_push(std::move(cmd));
       }
   }
   ```

   // Active codec indicator (locked decision: sender-side indicator)
   ImGui::SameLine();

       ImGui::TextColored(ImVec4(0.2f, 0.8f, 0.2f, 1.0f), "[FLAC]");

       ImGui::TextDisabled("[Vorbis]");

4. Add recording checkbox (REC-01, REC-02). After the local channel controls section (e.g., after volume/pan/mute controls), add:
   ```cpp
   // Session recording (REC-01)
   if (ImGui::Checkbox("Record Session", &state.recording_enabled)) {
       // Set config_savelocalaudio: 2 = OGG + WAV, 0 = off
       std::unique_lock<std::mutex> lock(plugin->client_mutex);
       if (plugin->client) {
           plugin->client->config_savelocalaudio = state.recording_enabled ? 2 : 0;
       }
   }
   if (state.recording_enabled) {
       ImGui::SameLine();
       ImGui::TextDisabled("(OGG + WAV)");
   }
   ```
   Note: config_savelocalaudio is not an atomic and is accessed from the Run thread, so protect with client_mutex. Alternatively, since it's just an int being set, and the Run thread reads it, the mutex approach is safest. However, if this is called from the UI render thread and client_mutex is already held elsewhere in the render path, check for deadlock. If client_mutex is only held briefly here, it should be safe.

1. Add FOURCC constants at top (same as ui_local.cpp).

2. After the channel name display (line ~106, after `ImGui::Text("%s", channel_label);`), add receiver-side codec indicator:
   ```cpp
   // Per-channel codec indicator (locked decision: receiver-side indicator)
   // Read codec_fourcc from UiRemoteChannel in ui_state
   ```
   The challenge: ui_remote.cpp currently reads channel state directly from NJClient via GetUserChannelState(). The codec_fourcc is in UiRemoteChannel but the render loop uses NJClient directly. We need to either:
   a) Read codec_fourcc from the UiState remote_users snapshot, or
   b) Call GetUserChannelCodec() on NJClient directly in the render loop

   Since the render loop already holds client_mutex and calls NJClient methods, option (b) is simpler:
   ```cpp
   // After channel_label display:
   unsigned int codec_fourcc = client->GetUserChannelCodec(u, channel_index);
   ImGui::SameLine();
   if (codec_fourcc == NJ_ENCODER_FMT_FLAC) {
       ImGui::TextColored(ImVec4(0.2f, 0.8f, 0.2f, 1.0f), "[FLAC]");
   } else if (codec_fourcc == MAKE_NJ_FOURCC('O','G','G','v')) {
       ImGui::TextDisabled("[Vorbis]");
   } else if (codec_fourcc != 0) {
       // Unsupported codec error indicator (locked decision)
       ImGui::TextColored(ImVec4(1.0f, 0.3f, 0.3f, 1.0f), "[Unsupported codec]");
   }
   ```
