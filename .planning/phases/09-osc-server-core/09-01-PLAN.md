#pragma once
#include <JuceHeader.h>
#include <array>
#include <string>

// Parameter entry types
enum class OscParamType
{
    ApvtsFloat,     // APVTS float param: use setValueNotifyingHost()
    ApvtsBool,      // APVTS bool param: use setValueNotifyingHost()
    NjclientAtomic, // NJClient atomic: direct store
    CmdQueue,       // Dispatch via cmd_queue (solo)
    ReadOnly        // Telemetry/VU: send-only
};

struct OscParamEntry
{
    juce::String oscAddress;     // e.g., "/JamWide/local/1/volume"
    juce::String apvtsId;        // e.g., "localVol_0" (empty for non-APVTS)
    OscParamType type;
    float rangeMin = 0.0f;       // OSC value range min
    float rangeMax = 1.0f;       // OSC value range max
    int channelIndex = -1;       // For local channels: 0-3; for cmd_queue solo
    bool isDbVariant = false;    // True for /volume/db addresses
};

class OscAddressMap
{
public:
    OscAddressMap();

    // Resolve an incoming OSC address to a param index. Returns -1 if not found.
    int resolve(const juce::String& address) const;

    // Get entry by index
    const OscParamEntry& getEntry(int index) const;

    // Get OSC address string for outgoing messages
    const juce::String& oscAddress(int index) const;

    // Get APVTS param ID for APVTS-backed params
    const juce::String& apvtsId(int index) const;

    // Total number of controllable parameters (OSC receive + send)
    int getControllableCount() const { return controllableCount; }

    // Total entries including read-only telemetry
    int getTotalCount() const { return static_cast<int>(entries.size()); }

    // Index ranges for iteration
    int getTelemetryStartIndex() const { return controllableCount; }

    // dB conversion helpers (per D-06: standard linear dB, not VbFader curve)
    static float linearToDb(float linear);
    static float dbToLinear(float db);

    // Pan conversion (APVTS -1..1 <-> OSC 0..1)
    static float apvtsPanToOsc(float apvtsPan);
    static float oscPanToApvts(float oscPan);

private:
    std::vector<OscParamEntry> entries;
    int controllableCount = 0;  // Entries before this index are controllable

    void buildMap();
};

**OscServer.cpp — implementation:**

**Constructor:** Initialize `lastSentValues` to all -999.0f (force first send). Initialize `oscSourced` to all false. Do NOT start timer or connect in constructor.

**`start(receivePort, sendIP, sendPort)`:**
1. Call `stop()` first to clean up any existing connection
2. Set `currentReceivePort`, `currentSendIP`, `currentSendPort`
3. Call `receiver.connect(receivePort)`. If returns false, set `receiverError = true`, `errorMsg = "Failed to bind port " + String(receivePort)`, return false
4. Call `receiver.addListener(this)` to register RealtimeCallback listener
5. Call `sender.connect(sendIP, sendPort)`. If returns false, receiver.disconnect(), set error, return false
6. Set `senderConnected = true`, `receiverConnected = true`, `receiverError = false`, `errorMsg = ""`
7. Set `enabled = true`
8. Call `startTimer(100)` — 100ms timer per D-12
9. Reset all `lastSentValues` to -999.0f and `oscSourced` to false (force full state dump on connect)
10. Return true

**`stop()`:**
1. `stopTimer()`
2. `receiver.removeListener(this)`, `receiver.disconnect()`
3. `sender.disconnect()`
4. Set `enabled = false`, `senderConnected = false`, `receiverConnected = false`

**`oscMessageReceived(msg)` — runs on juce_osc network thread (per D-19):**
1. Extract address string: `msg.getAddressPattern().toString()`
2. Extract value: if `msg.size() > 0 && msg[0].isFloat32()` use `getFloat32()`, else if `isInt32()` cast to float, else return
3. Capture address and value, dispatch to message thread:
```cpp
juce::MessageManager::callAsync([this, addr = address, val = value]() {
    handleOscOnMessageThread(addr, val);
});
```
This is CRITICAL per D-19: preserves SPSC single-producer invariant.

**`oscBundleReceived(bundle)`:** Iterate elements, call `oscMessageReceived` for messages, recurse for nested bundles.

**`handleOscOnMessageThread(address, value)` — runs on message thread:**
1. `int idx = addressMap.resolve(address);` — if -1, return (unknown address)
2. Get entry: `const auto& entry = addressMap.getEntry(idx);`
3. Clamp value to entry's range: `float clamped = juce::jlimit(entry.rangeMin, entry.rangeMax, value);`
4. Mark echo suppression: `oscSourced[idx] = true;` (per D-14)
5. Dispatch based on `entry.type`:

For `ApvtsFloat`:
- If `entry.isDbVariant`: convert from dB to linear: `float linear = OscAddressMap::dbToLinear(clamped);` then set APVTS: `float norm = param->getNormalisableRange().convertTo0to1(linear);` and `param->setValueNotifyingHost(norm);`
- If address contains `/pan`: convert OSC 0-1 to APVTS -1..1: `float apvtsPan = OscAddressMap::oscPanToApvts(clamped);` then `param->setValueNotifyingHost(param->getNormalisableRange().convertTo0to1(apvtsPan));`
- Otherwise: `param->setValueNotifyingHost(clamped);` (OSC 0-1 maps to APVTS normalized 0-1 directly)

For `ApvtsBool`:
- `param->setValueNotifyingHost(clamped >= 0.5f ? 1.0f : 0.0f);`

For `NjclientAtomic` (metro pan only, per Pattern 4):
