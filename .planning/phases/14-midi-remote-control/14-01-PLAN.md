```cpp
class JamWideJuceProcessor;

class MidiMapper : private juce::Timer
{
public:
    explicit MidiMapper(JamWideJuceProcessor& proc);
    ~MidiMapper();

    // Audio thread: parse incoming CC from processBlock MidiBuffer (per D-04)
    void processIncomingMidi(const juce::MidiBuffer& buffer);

    // Audio thread: append CC feedback to processBlock MidiBuffer (per D-11)

    // Message thread: mapping management
    bool addMapping(const juce::String& paramId, int ccNumber, int midiChannel);
    void removeMapping(const juce::String& paramId);
    void removeMappingByCc(int ccNumber, int midiChannel);
    void clearAllMappings();
    int getMappingCount() const;

    // Query
    bool hasMapping(const juce::String& paramId) const;
    bool hasMappingForCc(int ccNumber, int midiChannel) const;
    struct MappingInfo { juce::String paramId; int ccNumber; int midiChannel; };
    std::vector<MappingInfo> getAllMappings() const;
    std::optional<MappingInfo> getMappingForParam(const juce::String& paramId) const;

    // State persistence (per D-20)
    void saveToState(juce::ValueTree& state) const;
    void loadFromState(const juce::ValueTree& state);

    // Standalone device management (per D-05)
    void openMidiInput(const juce::String& deviceId);
    void openMidiOutput(const juce::String& deviceId);
    void closeMidiInput();
    void closeMidiOutput();

    // Cross-system echo suppression API (for OscServer/ChannelStripArea callers)
    // Suppresses MIDI feedback for the given paramId for one timer tick.
    // Addresses review concern: "Make echo suppression mapping-aware"
    void setEchoSuppression(const juce::String& paramId);

    // Remote slot reset (per D-17)
    void resetRemoteSlotDefaults(int slotIndex);

    // Status
    bool hasActiveMappings() const;
    bool isReceiving() const;
    bool hasError() const;

    // Error state semantics (addresses review concern: "Specify error semantics for footer status dot")
    // disabled = no mappings (grey dot)
    // healthy = mappings exist AND receiving CC within last 2s (green dot)
    // degraded = mappings exist but no recent CC (green dot, steady)
    // failed = standalone device open failure (red dot)
    enum class Status { Disabled, Healthy, Degraded, Failed };
    Status getStatus() const;

    // Cap per D-09: 16 channels x 128 CCs = 2048
    static constexpr int kMaxMappings = 2048;

private:
    void timerCallback() override; // Centralized APVTS-to-NJClient bridge + standalone feedback

    // Composite key: ((channel-1) << 7) | ccNumber, where channel is 1-based JUCE convention
    static int makeKey(int ccNumber, int midiChannel) { return ((midiChannel - 1) << 7) | ccNumber; }

    struct Mapping {
        juce::String paramId;
        int ccNumber;
        int midiChannel;
    };

    // Thread-safe map access: audio thread reads published_, message thread writes staging_
    // After modification on message thread, atomically swap pointer
    struct MappingTable {
        std::unordered_map<int, Mapping> ccToParam;      // key -> mapping
        std::unordered_map<juce::String, int> paramToCc;  // paramId -> key
    };

    std::shared_ptr<const MappingTable> published_;  // read by audio thread via atomic load
    MappingTable staging_;  // written by message thread only

    void publishMappings();  // copy staging_ to published_ (atomic swap)

    // Per-mapping echo suppression (addresses review: "mapping-aware, not global 2-frame window")
    // Key: composite MIDI key. Value: suppression counter.
    // When CC arrives from MIDI input: set to 2 (suppress current + next appendFeedbackMidi call)
    // When external caller (OSC/UI) sets suppression: set to 2
    // Decremented each appendFeedbackMidi call; feedback suppressed while > 0
    // This is per-(channel, cc) which handles varying block sizes correctly.
    std::unordered_map<int, int> echoSuppression_;

    // Dirty tracking for feedback (per research recommendation)
    std::unordered_map<int, int> lastSentCcValues_;  // key -> last CC value sent

    // APVTS-to-NJClient sync tracking for remote params
    // THIS IS THE SOLE PATH from APVTS remote params to NJClient cmd_queue.
    // Addresses review concern: "Centralize command emission in one observer/bridge"
    // Timer runs at 20ms (not 100ms) for responsive mixer control.
    // Addresses review concern: "100ms timer latency too slow for mixer control"
    std::array<float, 16> lastSyncedRemoteVol_{};
    std::array<float, 16> lastSyncedRemotePan_{};
    std::array<bool, 16>  lastSyncedRemoteMute_{};

    // Local solo and metro pan APVTS-to-NJClient sync
    std::array<bool, 4> lastSyncedLocalSolo_{};
    float lastSyncedMetroPan_ = 0.0f;

    // Activity tracking
    std::atomic<bool> receivingMidi_{false};
    std::atomic<int64_t> lastMidiReceivedTime_{0};

    // Standalone device state (per D-05)
    std::unique_ptr<juce::MidiInput> midiInput_;
    std::unique_ptr<juce::MidiOutput> midiOutput_;
    std::atomic<bool> deviceError_{false};

    JamWideJuceProcessor& processor;

    // Validation helpers
    static bool isValidCc(int cc) { return cc >= 0 && cc <= 127; }
    static bool isValidChannel(int ch) { return ch >= 1 && ch <= 16; }
};
```

`MidiMapper.cpp` implementation details:

- **Constructor**: Initialize published_ with empty MappingTable shared_ptr. Initialize lastSyncedRemoteVol_ to all 1.0f (default volume), lastSyncedRemotePan_ to all 0.0f (center), lastSyncedRemoteMute_ to all false (unmuted), lastSyncedLocalSolo_ to all false, lastSyncedMetroPan_ to 0.0f. Start timer at **20ms** (NOT 100ms -- addresses review concern about latency). 20ms (50Hz) gives near-immediate response for physical fader control while staying well within message thread timer precision.

- **Destructor**: stopTimer(). Close any open MIDI devices.

- **processIncomingMidi** (audio thread, per D-04):
  Load published_ via `std::atomic_load(&published_)`.
  Iterate MidiBuffer metadata. For each `msg.isController()`:
  - Extract cc = `msg.getControllerNumber()`, ch = `msg.getChannel()` (1-based), value = `msg.getControllerValue()`
  - Validate: cc 0-127, ch 1-16. Skip invalid.
  - Update receivingMidi_ and lastMidiReceivedTime_ with `juce::Time::currentTimeMillis()`.
  - If MidiLearnManager is learning (check via processor reference): call tryLearn with cc/ch. If tryLearn returns true (completed learning), skip normal dispatch for this message.
  - Build key = makeKey(cc, ch). Look up in table->ccToParam.
  - If found: get paramId. Look up APVTS parameter via `processor.apvts.getParameter(paramId)`.
  - For `AudioParameterBool` (mute/solo per D-08): if value > 0, toggle (read current, flip). If value == 0, ignore (button release).
  - For float params (vol/pan per D-10): `param->setValueNotifyingHost(value / 127.0f)`.
  - Set `echoSuppression_[key] = 2` (per-mapping suppression, per D-11).

- **appendFeedbackMidi** (audio thread):
  Load published_ via atomic_load.
  For each mapping in table->ccToParam:
  - If echoSuppression_[key] > 0: decrement, skip this mapping's feedback.
  - Read current param value via `processor.apvts.getParameter(paramId)->getValue()`.
  - For bool params: ccValue = (value >= 0.5f) ? 127 : 0 (per D-08 feedback).
  - For float params: ccValue = roundToInt(value * 127.0f) (per D-10).
  - Compare with lastSentCcValues_[key]. If different: add `MidiMessage::controllerEvent(ch, cc, ccValue)` to buffer at sample offset 0. Update lastSentCcValues_.
  Note: feedback only appends to existing buffer (never clears it). Non-CC MIDI events in the buffer are preserved.

- **addMapping** (message thread, per D-09):
  Validate cc 0-127, channel 1-16. Check count < kMaxMappings (2048). Return false if invalid or at capacity.
  Duplicate conflict handling (addresses review concern): If the CC+Ch key already maps to a different paramId, remove that old mapping first (last-write-wins per UI-SPEC "MIDI Learn Conflict Resolution"). If the paramId already has a mapping to a different CC+Ch, remove the old CC+Ch mapping. Add to staging_.ccToParam and staging_.paramToCc. Call publishMappings(). Return true.

- **removeMapping**: Remove from staging_ by paramId lookup. publishMappings().

- **removeMappingByCc**: Remove from staging_ by key lookup. publishMappings().

- **clearAllMappings**: Clear staging_.ccToParam and staging_.paramToCc. publishMappings(). Reset lastSentCcValues_ and echoSuppression_.

- **publishMappings**: Create new `shared_ptr<const MappingTable>` from staging_ copy. `std::atomic_store(&published_, newTable)`.

- **saveToState** (per D-20): Create ValueTree child "MidiMappings". For each mapping in staging_: create child "Mapping" with properties "paramId" (String), "cc" (int), "channel" (int). Add to parent state.

- **loadFromState** (per D-20): Find child "MidiMappings". Clear staging_. For each "Mapping" child: validate cc (clamp 0-127 via `juce::jlimit(0, 127, rawCc)`), validate channel (clamp 1-16 via `juce::jlimit(1, 16, rawCh)`), check paramId is not empty, check paramId exists in processor.apvts (via `getParameter(paramId) != nullptr`). Skip entries that fail validation entirely (don't add malformed entries). Add valid entries to staging_. publishMappings().

  Addresses review concern: "Add tests for malformed saved mappings" -- Test 13 covers out-of-range CC, invalid channel, empty paramId.

- **setEchoSuppression** (message thread, for OscServer/ChannelStripArea callers):
  Look up paramId in published_ table's paramToCc map. If found, set echoSuppression_[key] = 2.
  Addresses review concern: "Make echo suppression mapping-aware: (channel, cc, direction, lastValue/time) rather than global 2-frame window"

- **timerCallback** (message thread, **20ms** interval):

  ```cpp
  void MidiMapper::timerCallback()
  {
      // === CENTRALIZED APVTS-to-NJClient remote sync (SOLE cmd_queue path) ===
      const int count = juce::jmin(processor.userCount.load(std::memory_order_relaxed), 16);
      for (int i = 0; i < count; ++i)
      {
          juce::String suffix = juce::String(i);
          float vol = *processor.apvts.getRawParameterValue("remoteVol_" + suffix);
          float pan = *processor.apvts.getRawParameterValue("remotePan_" + suffix);
          bool mute = *processor.apvts.getRawParameterValue("remoteMute_" + suffix) >= 0.5f;

          bool volChanged = std::abs(vol - lastSyncedRemoteVol_[i]) > 0.001f;
          bool panChanged = std::abs(pan - lastSyncedRemotePan_[i]) > 0.001f;
          bool muteChanged = mute != lastSyncedRemoteMute_[i];

          if (volChanged || panChanged || muteChanged)
          {
              jamwide::SetUserStateCommand cmd;
              cmd.user_index = i;
              cmd.set_vol = volChanged;
              cmd.volume = vol;
              cmd.set_pan = panChanged;
              cmd.pan = pan;
              cmd.set_mute = muteChanged;
              cmd.mute = mute;
              processor.cmd_queue.push(std::move(cmd));

              if (volChanged) lastSyncedRemoteVol_[i] = vol;
              if (panChanged) lastSyncedRemotePan_[i] = pan;
              if (muteChanged) lastSyncedRemoteMute_[i] = mute;
          }
      }

          }
      }

      // === Metro pan APVTS-to-NJClient sync (D-16) ===
      float metroPan = *processor.apvts.getRawParameterValue("metroPan");
      if (std::abs(metroPan - lastSyncedMetroPan_) > 0.001f)
      {
          lastSyncedMetroPan_ = metroPan;
          // Write directly to NJClient atomic (same as existing OscServer metro pan path)
          if (auto* client = processor.getClient())
              client->config_metronome_pan = metroPan;
      }

      // === Standalone MIDI output feedback ===
      if (midiOutput_)
      {
          auto table = std::atomic_load(&published_);
          if (table)
          {
              for (const auto& [key, mapping] : table->ccToParam)
              {
                  // Check per-mapping echo suppression
                  auto suppIt = echoSuppression_.find(key);
                  if (suppIt != echoSuppression_.end() && suppIt->second > 0)
                  {
                      suppIt->second--;
                      continue;
                  }

                  auto* param = processor.apvts.getParameter(mapping.paramId);
                  if (!param) continue;

                  float currentNorm = param->getValue();
                  int ccValue;
                  if (dynamic_cast<juce::AudioParameterBool*>(param))
                      ccValue = (currentNorm >= 0.5f) ? 127 : 0;
                  else
                      ccValue = juce::roundToInt(currentNorm * 127.0f);

                  if (ccValue != lastSentCcValues_[key])
                  {
                      lastSentCcValues_[key] = ccValue;
                      int channel = (key >> 7) + 1;
                      int cc = key & 0x7F;
                      midiOutput_->sendMessageNow(
                          juce::MidiMessage::controllerEvent(channel, cc, ccValue));
                  }
              }
          }
      }
  }
  ```
