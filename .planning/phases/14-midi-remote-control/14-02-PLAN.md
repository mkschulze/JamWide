  juce::String MidiConfigDialog::getRangeTextForParam(const juce::String& paramId)
  {
      // Volume parameters: range is 0-2 (linear)
      if (paramId.contains("Vol"))

      // Pan parameters: range is -1 to +1
      if (paramId.contains("Pan"))
          return juce::CharPointer_UTF8("\xc2\xb1") + juce::String("1 pan");  // +/-1 pan
