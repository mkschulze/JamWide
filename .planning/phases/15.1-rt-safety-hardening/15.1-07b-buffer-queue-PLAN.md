        TEST("M-7 bounds-check: oversized sample_count is rejected at the helper, increments drop counter");
        jamwide::SpscRing<jamwide::BlockRecord, 16> ring;
        std::atomic<uint64_t> drops{0};

        auto push = [&](int sc, int nch, const float* samples) {

                || sc <= 0 || nch <= 0) {
