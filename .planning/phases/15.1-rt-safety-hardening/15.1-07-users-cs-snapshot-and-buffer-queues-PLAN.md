    ```cpp
    int DecodeMediaBuffer::Read(void* buf, int len)
    {
        int written = 0;
        uint8_t* out = static_cast<uint8_t*>(buf);
        while (written < len) {
            // 1. Drain consumer-side buffer first.
            int avail = m_consumer_buf_len - m_consumer_buf_pos;
            if (avail > 0) {
                int take = std::min(avail, len - written);
                std::memcpy(out + written, m_consumer_buf.data() + m_consumer_buf_pos, take);
                m_consumer_buf_pos += take;
                written += take;
                continue;
            }
            // 2. Pull next chunk from SPSC.
            auto chunk = m_chunks.try_pop();
            if (!chunk) break;  // empty — return what we have
            std::memcpy(m_consumer_buf.data(), chunk->data, chunk->len);
            m_consumer_buf_len = chunk->len;
            m_consumer_buf_pos = 0;
        }
        return written;
    }
    ```

    ```cpp
    int DecodeMediaBuffer::Write(const void* data, int len)

        // Caller may need to split into chunks if len > CHUNK_BYTES.
        const uint8_t* in = static_cast<const uint8_t*>(data);
        int remaining = len;
        while (remaining > 0) {
    ```
