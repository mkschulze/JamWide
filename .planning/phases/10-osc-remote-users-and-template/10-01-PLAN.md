**3. Implement handleRemoteUserOsc:**

Parse the OSC address to extract user index and control name. Two patterns:
- Group bus: `/JamWide/remote/{idx}/volume` (or `/pan`, `/mute`, `/volume/db`)
- Sub-channel: `/JamWide/remote/{idx}/ch/{n}/volume` (or `/pan`, `/mute`, `/solo`, `/volume/db`)

Implementation:
```cpp
void OscServer::handleRemoteUserOsc(const juce::String& address, float value)
{
    // Parse: /JamWide/remote/{idx}/...
    // Strip prefix "/JamWide/remote/" (16 chars)
    juce::String remainder = address.substring(16);

    // Extract idx (1-based in OSC, per D-02)
    int slashPos = remainder.indexOf("/");
    if (slashPos < 0) return;

    int oscIdx = remainder.substring(0, slashPos).getIntValue();
    if (oscIdx < 1 || oscIdx > kMaxRemoteSlots) return;  // bounds check per T-10-02

    // Check user exists in current snapshot
    const auto& users = processor.cachedUsers;
    int userIndex = oscIdx - 1;  // convert to 0-based (Pitfall 1)
    if (userIndex >= static_cast<int>(users.size())) return;

    juce::String control = remainder.substring(slashPos + 1);  // e.g., "volume", "ch/1/volume"

    // Sub-channel path: starts with "ch/"
    if (control.startsWith("ch/"))
    {
        // Parse: ch/{n}/{control}
        juce::String chRemainder = control.substring(3);  // after "ch/"
        int chSlash = chRemainder.indexOf("/");
        if (chSlash < 0) return;

        int oscChIdx = chRemainder.substring(0, chSlash).getIntValue();
        if (oscChIdx < 1) return;
        int seqIdx = oscChIdx - 1;  // 0-based sequential index

        // Resolve NINJAM bit index from sequential index (Pitfall 2)
        const auto& user = users[static_cast<size_t>(userIndex)];
        if (seqIdx >= static_cast<int>(user.channels.size())) return;
        int njChannelIdx = user.channels[static_cast<size_t>(seqIdx)].channel_index;

        juce::String chControl = chRemainder.substring(chSlash + 1);

        jamwide::SetUserChannelStateCommand cmd;
        cmd.user_index = userIndex;
        cmd.channel_index = njChannelIdx;

        if (chControl == "volume")
        {
            cmd.set_vol = true;
            cmd.volume = juce::jlimit(0.0f, 1.0f, value) * 2.0f;  // OSC 0-1 -> NJClient 0-2
        }
        else if (chControl == "volume/db")
        {
            cmd.set_vol = true;
            cmd.volume = OscAddressMap::dbToLinear(juce::jlimit(-100.0f, 6.0f, value));
        }
        else if (chControl == "pan")
        {
            cmd.set_pan = true;
            cmd.pan = OscAddressMap::oscPanToApvts(juce::jlimit(0.0f, 1.0f, value));
        }
        else if (chControl == "mute")
        {
            cmd.set_mute = true;
            cmd.mute = (value >= 0.5f);
        }
        else if (chControl == "solo")
        {
            cmd.set_solo = true;
            cmd.solo = (value >= 0.5f);
        }
        else return;  // Unknown sub-channel control

        processor.cmd_queue.try_push(std::move(cmd));
    }
    else
    {
        // Group bus path: volume, pan, mute, solo
        // Per Research Open Question 2: group-level solo dispatches SetUserChannelStateCommand
        // with set_solo on ALL sub-channels (no solo field on SetUserState).

        if (control == "volume")
        {
            jamwide::SetUserStateCommand cmd;
            cmd.user_index = userIndex;
            cmd.set_vol = true;
            cmd.volume = juce::jlimit(0.0f, 1.0f, value) * 2.0f;
            processor.cmd_queue.try_push(std::move(cmd));
        }
        else if (control == "volume/db")
        {
            jamwide::SetUserStateCommand cmd;
            cmd.user_index = userIndex;
            cmd.set_vol = true;
            cmd.volume = OscAddressMap::dbToLinear(juce::jlimit(-100.0f, 6.0f, value));
            processor.cmd_queue.try_push(std::move(cmd));
        }
        else if (control == "pan")
        {
            jamwide::SetUserStateCommand cmd;
            cmd.user_index = userIndex;
            cmd.set_pan = true;
            cmd.pan = OscAddressMap::oscPanToApvts(juce::jlimit(0.0f, 1.0f, value));
            processor.cmd_queue.try_push(std::move(cmd));
        }
        else if (control == "mute")
        {
            jamwide::SetUserStateCommand cmd;
            cmd.user_index = userIndex;
            cmd.set_mute = true;
            cmd.mute = (value >= 0.5f);
            processor.cmd_queue.try_push(std::move(cmd));
        }
        else if (control == "solo")
        {
            // Group solo: set solo on ALL sub-channels (Research Open Question 2)
            const auto& user = users[static_cast<size_t>(userIndex)];
            bool soloState = (value >= 0.5f);
            for (const auto& ch : user.channels)
            {
                jamwide::SetUserChannelStateCommand cmd;
                cmd.user_index = userIndex;
                cmd.channel_index = ch.channel_index;
```
